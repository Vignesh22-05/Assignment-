import tkinter as tk
from tkinter import messagebox

# Function to append text to file
def append_to_file():
    text = entry.get()
    if text:  # check if text is not empty
        with open("sample.txt", "a") as file:
            file.write(text + "\n")
        entry.delete(0, tk.END)
        update_display()  # update the display after appending text
    else:
        messagebox.showwarning("Input Error", "Please enter some text.")

# Function to update the display area with file content
def update_display():
    try:
        with open("sample.txt", "r") as file:
            content = file.read()
            text_area.delete(1.0, tk.END)  # clear the text area
            text_area.insert(tk.END, content)  # insert updated content
    except FileNotFoundError:
        # If the file does not exist, create an empty one
        with open("sample.txt", "w") as file:
            pass
        text_area.delete(1.0, tk.END)

# Creating the main window
root = tk.Tk()
root.title("Text Appender")
root.geometry("400x300")

# Entry box to get text from user
entry_label = tk.Label(root, text="Enter text:")
entry_label.pack(pady=5)

entry = tk.Entry(root, width=40)
entry.pack(pady=5)

# Button to append the text to file
append_button = tk.Button(root, text="Append to File", command=append_to_file)
append_button.pack(pady=10)

# Text area to display the contents of the file
text_area = tk.Text(root, width=40, height=10)
text_area.pack(pady=10)

# Initialize the text area with the content of the file
update_display()

# Run the application
root.mainloop()
